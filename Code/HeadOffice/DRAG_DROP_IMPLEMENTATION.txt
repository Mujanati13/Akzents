// Get Vorher images
getVorherImages(photoIndex: number): ImageItem[] {
  if (photoIndex === 0) return this.prepopulatedVorherImages1;
  else if (photoIndex === 2) return this.prepopulatedVorherImages3;
  return [];
}

// Get Nachher images  
getNachherImages(photoIndex: number): ImageItem[] {
  if (photoIndex === 0) return this.prepopulatedNachherImages1;
  else if (photoIndex === 2) return this.prepopulatedNachherImages3;
  return [];
}

// Get Nachher only images
getNachherOnlyImages(photoIndex: number): ImageItem[] {
  if (photoIndex === 1) return this.prepopulatedNachherImages2;
  return [];
}

// Handle Vorher drop
onVorherDropped(event: CdkDragDrop<ImageItem[]>, photoIndex: number): void {
  if (event.previousContainer === event.container) {
    moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
  }
}

// Handle Nachher drop
onNachherDropped(event: CdkDragDrop<ImageItem[]>, photoIndex: number): void {
  if (event.previousContainer === event.container) {
    moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
  }
}

// Handle Nachher only drop
onNachherOnlyDropped(event: CdkDragDrop<ImageItem[]>, photoIndex: number): void {
  if (event.previousContainer === event.container) {
    moveItemInArray(event.container.data, event.previousIndex, event.currentIndex);
  }
}
